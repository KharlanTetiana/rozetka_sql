Вывести "EMPLOYEE NAME", DOB, "DEPARTMENT NAME" of "QA" DEPARTMENT, Sorting by "EMPLOYEE NAME" ASC

SELECT  EMPNAME, DOB,  DEPTNAME
FROM   EMP as A
JOIN DEPT as B
ON A.DEPTID=B.DEPTID
ORDER BY EMPNAME ASC
Вывести "EMPLOYEE NAME", "AGE", "DEPARTMENT NAME" Sort by "AGE" DESC

SELECT EMPNAME as EMPLOYEE NAME, DEPTNAME as  DEPARTMENT NAME, DATEDIFF (yy, DOB, DateTime.NOW)as AGE
FROM EMP as A
JOIN DEPT as B
ON A.DEPID=B.DEPID
ORDER BY AGE DESC

Вывести "DEPARTMENT NAME", Number of Employees Sort by Number of Employees

SELECT DEPTNAME as DEPARTMENT NAME, count (EMPNAME) as Number of Employees
FROM  DEPT as B
JOIN   IMP as A
ON  A.DEPTID=B.DEPTID
GROUP BY  DEPTNAME
ORDER BY Number of Employees

Вывести EMPLOYEE NAME, MANAGER NAME, DEPARTMENT NAME
SELECT EMPNAME as EMPLOYEE NAME, MGRID, DEPTNAME as  DEPARTMENT NAME
FROM  EMP as A
LEFT JOIN   DEPT as B
ON A.DEPTID=B.DEPTID

Вывести EMPLOYEE NAME, MANAGER NAME, DEPARTMENT NAME

SELECT EMPNAME as EMPLOYEE NAME, MGRID, DEPTNAME as DEPARTMENT NAME, count(EMPID)
FROM  EMP as A
JOIN  DEPT as B
ON A.DEPTID=B.DEPTID
GROUP BY  DEPTNAME
HAVING count(EMPID) > 1
